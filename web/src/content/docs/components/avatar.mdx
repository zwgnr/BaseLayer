## Installation

Install via <a href="/docs/getting-started/cli">CLI</a> or copy and paste the potion below into a new file, preferably in a folder called *potions*.

## Potion 

<Potion>
<Fragment slot="contentSlot">
```tsx
import { tv } from "tailwind-variants";

export const avatarPotion = tv({
  slots: {
    avatarRoot: "h-16 w-16 rounded-full border-2 border-accent",
    avatarFallback:
      "flex h-[inherit] items-center justify-center bg-surface-2 font-bold text-text",
    avatarImage: "object-cover",
  },
});

```

```tsx
import {
  AvatarFallback,
  AvatarImage,
  Avatar as AvatarRoot,
} from "@ark-ui/react";
import { avatarPotion } from "@potions/avatarPotion";

const { avatarRoot, avatarFallback, avatarImage } = avatarPotion();

export const Avatar = () => (
  <AvatarRoot className={avatarRoot()}>
    <AvatarFallback className={avatarFallback()}>ME</AvatarFallback>
    <AvatarImage
      className={avatarImage()}
      src="https://i.pravatar.cc/300"
      alt="avatar"
    />
  </AvatarRoot>
);

```
