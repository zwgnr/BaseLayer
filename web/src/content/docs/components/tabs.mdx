## Manual Install

<p>
  Ensure Ark-UI is installed as well as Tailwind Variants. Then simply copy and
  paste the code below into a new file!
</p>

<Potion>
<Fragment slot="contentSlot">
```tsx
import {
  Radio as ArkRadio,
  RadioControl as ArkRadioControl,
  RadioGroup as ArkRadioGroup,
  RadioInput as ArkRadioInput,
  RadioLabel as ArkRadioLabel,
} from "@ark-ui/react";

import { ComponentPropsWithoutRef, ElementRef, forwardRef } from "react";
import { tv } from "tailwind-variants";

const radioGroupStyles = tv({
  slots: {
    root: "flex flex-col gap-2",
    radioControl:
      "h-5 w-5 rounded-full border-2 bg-surface data-[checked]:border-primary data-[checked]:bg-surface data-[checked]:outline data-[checked]:outline-[4px] data-[checked]:outline-offset-[-5px] data-[checked]:outline-primary",
    radio: "flex cursor-pointer items-center gap-3",
    radioLabel: "",
  },
});

const { root, radioControl, radio, radioLabel } = radioGroupStyles();

const RadioGroup = forwardRef<
  ElementRef<typeof ArkRadioGroup>,
  ComponentPropsWithoutRef<typeof ArkRadioGroup>
>((props, ref) => {
  const { className, children, ...rest } = props;
  return (
    <ArkRadioGroup ref={ref} className={root({ className })} {...rest}>
      {children}
    </ArkRadioGroup>
  );
});

RadioGroup.displayName = ArkRadioGroup.displayName;

const Radio = forwardRef<
  ElementRef<typeof ArkRadio>,
  ComponentPropsWithoutRef<typeof ArkRadio>
>((props, ref) => {
  const { className, children, ...rest } = props;
  return (
    <ArkRadio ref={ref} className={radio({ className })} {...rest}>
      {children}
    </ArkRadio>
  );
});

Radio.displayName = ArkRadio.displayName;

const RadioInput = forwardRef<
  ElementRef<typeof ArkRadioInput>,
  ComponentPropsWithoutRef<typeof ArkRadioInput>
>((props, ref) => {
  const { className, children, ...rest } = props;
  return (
    <ArkRadioInput ref={ref} className="out peer" {...rest}>
      {children}
    </ArkRadioInput>
  );
});

RadioInput.displayName = ArkRadioInput.displayName;

const RadioControl = forwardRef<
  ElementRef<typeof ArkRadioControl>,
  ComponentPropsWithoutRef<typeof ArkRadioControl>
>((props, ref) => {
  const { className, children, ...rest } = props;
  return (
    <ArkRadioControl
      ref={ref}
      className={radioControl({ className })}
      {...rest}
    />
  );
});

RadioControl.displayName = ArkRadioControl.displayName;

const RadioLabel = forwardRef<
  ElementRef<typeof ArkRadioLabel>,
  ComponentPropsWithoutRef<typeof ArkRadioLabel>
>((props, ref) => {
  const { className, children, ...rest } = props;
  return (
    <ArkRadioLabel ref={ref} className={radioLabel({ className })} {...rest}>
      {children}
    </ArkRadioLabel>
  );
});

RadioLabel.displayName = ArkRadioLabel.displayName;

export { Radio, RadioControl, RadioGroup, RadioInput, RadioLabel };


```
</Fragment>
</Potion>
