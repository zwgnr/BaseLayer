## Manual Install

<p>
  Ensure Ark-UI is installed as well as Tailwind Variants. Then simply copy and
  paste the code below into a new file!
</p>

<Potion>
<Fragment slot="contentSlot">
```tsx
import {
  Carousel as ArkCarousel,
  CarouselControl as ArkCarouselControl,
  CarouselNextSlideTrigger as ArkCarouselNextSlideTrigger,
  CarouselPrevSlideTrigger as ArkCarouselPrevSlideTrigger,
  CarouselSlide as ArkCarouselSlide,
  CarouselSlideGroup as ArkCarouselSlideGroup,
  CarouselViewport as ArkCarouselViewport,
} from "@ark-ui/react";
import { ComponentPropsWithoutRef, ElementRef, forwardRef } from "react";
import { tv } from "tailwind-variants";

const carouselStyles = tv({
  slots: {
    root: "w-full",
    control: "flex absolute justify-between left-4 right-4 top-1/2 z-1 -translate-y-2/4	",
  },
});

const { root, control } = carouselStyles();

const Carousel = forwardRef<
  ElementRef<typeof ArkCarousel>,
  ComponentPropsWithoutRef<typeof ArkCarousel>
>(({ className, children, ...props }, ref) => (
  <ArkCarousel ref={ref} className={root({ className: className })} {...props}>
    <ArkCarouselViewport className="relative overflow-x-hidden">
      {children}
    </ArkCarouselViewport>
  </ArkCarousel>
));

Carousel.displayName = ArkCarousel.displayName;

const CarouselControl = forwardRef<
  ElementRef<typeof ArkCarouselControl>,
  ComponentPropsWithoutRef<typeof ArkCarouselControl>
>(({ className, children, ...props }, ref) => (
  <ArkCarouselControl
    ref={ref}
    className={control({ className: className })}
    {...props}
  >
    {children}
  </ArkCarouselControl>
));

CarouselControl.displayName = ArkCarouselControl.displayName;

const CarouselPrevSlideTrigger = ArkCarouselPrevSlideTrigger;
CarouselPrevSlideTrigger.displayName = ArkCarouselPrevSlideTrigger.displayName;

const CarouselNextSlideTrigger = ArkCarouselNextSlideTrigger;
CarouselNextSlideTrigger.displayName = ArkCarouselNextSlideTrigger.displayName;

const CarouselSlideGroup = ArkCarouselSlideGroup

CarouselSlideGroup.displayName = ArkCarouselSlideGroup.displayName;

const CarouselSlide = ArkCarouselSlide;
CarouselSlide.displayName = ArkCarouselSlide.displayName;

export {
  Carousel,
  CarouselControl,
  CarouselPrevSlideTrigger,
  CarouselNextSlideTrigger,
  CarouselSlideGroup,
  CarouselSlide,
};

```
</Fragment>
