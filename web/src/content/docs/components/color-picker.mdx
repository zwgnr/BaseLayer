```tsx
import {
  ColorPicker,
  ColorPickerArea,
  ColorPickerChannelInput,
  ColorPickerChannelSliderTrack,
  ColorPickerContent,
  ColorPickerEyeDropperTrigger,
  ColorPickerSwatch,
  ColorPickerSwatchGroup,
} from "@components/ColorPicker";
import { Pipette } from "lucide-react";

const presets = [
  "hsl(10, 81%, 59%)",
  "hsl(60, 81%, 59%)",
  "hsl(100, 81%, 59%)",
  "hsl(175, 81%, 59%)",
  "hsl(190, 81%, 59%)",
  "hsl(205, 81%, 59%)",
  "hsl(220, 81%, 59%)",
  "hsl(250, 81%, 59%)",
  "hsl(280, 81%, 59%)",
  "hsl(350, 81%, 59%)",
];

export const ColorPickerDemo = () => {
  return (
    <ColorPicker defaultValue="hsl(10, 81%, 59%)">
      {(api) => {
        const [hue, saturation, lightness] = api.channels;
        return (
          <ColorPickerContent>
            <ColorPickerArea xChannel={saturation} yChannel={lightness} />
            <div className="flex w-full flex-row gap-4 p-4">
              <ColorPickerEyeDropperTrigger>
                <Pipette />
              </ColorPickerEyeDropperTrigger>
              <div className="flex w-full flex-col gap-2">
                <ColorPickerChannelSliderTrack channel={hue} />
                <ColorPickerChannelSliderTrack channel="alpha" />
              </div>
            </div>
            <div className="flex justify-center gap-2 p-4 pb-0">
              <ColorPickerChannelInput channel={hue} />
              <ColorPickerChannelInput channel={saturation} />
              <ColorPickerChannelInput channel={lightness} />
              <ColorPickerChannelInput channel="alpha" />
            </div>
            <div className="p-4">
              <p>Presets</p>
              <ColorPickerSwatchGroup>
                {presets.map((color) => (
                  <ColorPickerSwatch key={color} value={color} />
                ))}
              </ColorPickerSwatchGroup>
            </div>
          </ColorPickerContent>
        );
      }}
    </ColorPicker>
  );
};
```

## Manual Install

<p>
  Ensure Ark-UI is installed as well as Tailwind Variants. Then simply copy and
  paste the code below into a new file!
</p>

<Potion>
<Fragment slot="contentSlot">
```tsx
import {
  ColorPicker as ArkColorPicker,
  ColorPickerArea as ArkColorPickerArea,
  ColorPickerAreaGradient as ArkColorPickerAreaGradient,
  ColorPickerAreaThumb as ArkColorPickerAreaThumb,
  ColorPickerChannelInput as ArkColorPickerChannelInput,
  ColorPickerChannelSliderBackground as ArkColorPickerChannelSliderBackground,
  ColorPickerChannelSliderThumb as ArkColorPickerChannelSliderThumb,
  ColorPickerChannelSliderTrack as ArkColorPickerChannelSliderTrack,
  ColorPickerContent as ArkColorPickerContent,
  ColorPickerEyeDropperTrigger as ArkColorPickerEyeDropperTrigger,
  ColorPickerSwatch as ArkColorPickerSwatch,
  ColorPickerSwatchBackground as ArkColorPickerSwatchBackground,
  ColorPickerSwatchGroup as ArkColorPickerSwatchGroup,
} from "@ark-ui/react";
import { ComponentPropsWithoutRef, ElementRef, forwardRef } from "react";
import { tv } from "tailwind-variants";

const colorPickerStyles = tv({
  slots: {
    content: "bg-surface rounded-xl border shadow-lg w-fit overflow-hidden",
    areaThumb:
      "rounded-full border-2 border-white h-4 w-4 shadow-lg shadow-black outline-none z-1",
    channelSliderTrack: "h-4 rounded-full border",
    swatch: "h-4 w-4 rounded-sm shrink-0",
    swatchGroup: "inline-flex flex-wrap gap-2 w-full",
    input: "border border-gray-200 rounded-md w-16",
    channelSliderThumb:
      "rounded-full border-2 border-white h-4 w-4 outline-none -translate-x-2/4 -translate-y-2/4 shadow-lg shadow-black",
  },
});

const {
  content,
  areaThumb,
  input,
  channelSliderTrack,
  swatch,
  swatchGroup,
  channelSliderThumb,
} = colorPickerStyles();

const ColorPicker = ArkColorPicker;
//
//ColorPicker.displayName = ArkColorPicker.displayName;

const ColorPickerContent = forwardRef<
  ElementRef<typeof ArkColorPickerContent>,
  ComponentPropsWithoutRef<typeof ArkColorPickerContent>
>(({ className, children, ...props }, ref) => (
  <ArkColorPickerContent
    ref={ref}
    className={content({ className: className })}
    {...props}
  >
    {children}
  </ArkColorPickerContent>
));

ColorPickerContent.displayName = ArkColorPickerContent.displayName;

const ColorPickerArea = forwardRef<
  ElementRef<typeof ArkColorPickerArea>,
  ComponentPropsWithoutRef<typeof ArkColorPickerArea>
>(({ className, children, ...props }, ref) => (
  <ArkColorPickerArea ref={ref} className="h-48" {...props}>
    <ArkColorPickerAreaGradient className="h-full" />
    <ArkColorPickerAreaThumb className={areaThumb({ className: className })} />
  </ArkColorPickerArea>
));

ColorPickerArea.displayName = ArkColorPickerArea.displayName;

const ColorPickerChannelSliderTrack = forwardRef<
  ElementRef<typeof ArkColorPickerChannelSliderTrack>,
  ComponentPropsWithoutRef<typeof ArkColorPickerChannelSliderTrack>
>(({ className, children, ...props }, ref) => (
  <ArkColorPickerChannelSliderTrack
    ref={ref}
    className={channelSliderTrack({ className: className })}
    {...props}
  >
    <ArkColorPickerChannelSliderBackground />
    <ArkColorPickerChannelSliderThumb className={channelSliderThumb()} />
  </ArkColorPickerChannelSliderTrack>
));

ColorPickerChannelSliderTrack.displayName =
  ArkColorPickerChannelSliderTrack.displayName;

const ColorPickerSwatchGroup = forwardRef<
  ElementRef<typeof ArkColorPickerSwatchGroup>,
  ComponentPropsWithoutRef<typeof ArkColorPickerSwatchGroup>
>(({ className, children, ...props }, ref) => (
  <ArkColorPickerSwatchGroup
    ref={ref}
    className={swatchGroup({ className: className })}
    {...props}
  >
    {children}
  </ArkColorPickerSwatchGroup>
));

ColorPickerSwatchGroup.displayName = ArkColorPickerSwatchGroup.displayName;

const ColorPickerSwatch = forwardRef<
  ElementRef<typeof ArkColorPickerSwatch>,
  ComponentPropsWithoutRef<typeof ArkColorPickerSwatch>
>(({ className, children, ...props }, ref) => (
  <ArkColorPickerSwatch
    ref={ref}
    className={swatch({ className: className })}
    {...props}
  >
    <ArkColorPickerSwatchBackground />
  </ArkColorPickerSwatch>
));

ColorPickerSwatch.displayName = ArkColorPickerSwatch.displayName;

const ColorPickerChannelInput = forwardRef<
  ElementRef<typeof ArkColorPickerChannelInput>,
  ComponentPropsWithoutRef<typeof ArkColorPickerChannelInput>
>(({ className, children, ...props }, ref) => (
  <ArkColorPickerChannelInput
    ref={ref}
    className={input({ className: className })}
    {...props}
  >
    {children}
  </ArkColorPickerChannelInput>
));

ColorPickerChannelInput.displayName = ArkColorPickerChannelInput.displayName;

const ColorPickerEyeDropperTrigger = ArkColorPickerEyeDropperTrigger;
ColorPickerEyeDropperTrigger.displayName =
  ArkColorPickerEyeDropperTrigger.displayName;

export {
  ColorPicker,
  ColorPickerContent,
  ColorPickerArea,
  ColorPickerChannelSliderTrack,
  ColorPickerSwatchGroup,
  ColorPickerSwatch,
  ColorPickerChannelInput,
  ColorPickerEyeDropperTrigger,
};


```
