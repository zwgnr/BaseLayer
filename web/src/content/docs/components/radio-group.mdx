import { button } from "@/components/base/button";
import { dialogPotion } from "@/components/base/dialogPotion";
import {
  Pressable as Button,
  DialogBackdrop,
  DialogCloseTrigger,
  DialogContainer,
  DialogContent,

  DialogTitle,
  DialogTrigger,
  Portal,

const { backdrop, container, content, description, title } = dialogPotion();

export const Dialog = () => {
  return (
    <DialogRoot>
      <DialogTrigger asChild>
        <Button className={button()}>Open dialog</Button>
      </DialogTrigger>
      <Portal>
        <DialogBackdrop className={backdrop()} />
        <DialogContainer className={container()}>
          <DialogContent className={content()}>
            <div className="flex flex-col gap-8 p-6">

                <DialogDescription className={description()}>
                  Dialog Description
                </DialogDescription>
              </div>
              <div className="flex w-full items-center justify-center gap-3">
                <DialogCloseTrigger asChild>
                  <Button
                    className={button({
                      state: "outline",
                      className:
                        "border-critical text-critical hover:bg-critical hover:text-black",
                    })}
                  >
                    Cancel
                  </Button>
                </DialogCloseTrigger>
                <Button className={button()}>Confirm</Button>
              </div>
            </div>
            <DialogCloseTrigger asChild>
              <Button className="absolute right-6 top-6">

              </Button>
            </DialogCloseTrigger>
          </DialogContent>
        </DialogContainer>
      </Portal>
    </DialogRoot>
  );
};
