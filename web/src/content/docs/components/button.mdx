```tsx
import { button, type buttonProps } from "@/potions/button";
import type { ButtonHTMLAttributes } from "react";

export interface ButtonProps
  extends ButtonHTMLAttributes<HTMLButtonElement>,
    buttonProps {}

export const Button = ({
  children,
  className,
  intent,
  size,
  state,
  ...props
}: ButtonProps) => {
  return (
    <button
      className={button({ intent, size, state, className })}
      {...props}
    >
      {children}
    </button>
  );
};

```
</Fragment>
